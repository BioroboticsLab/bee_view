284  /* http://psgraphics.blogspot.de/2011/01/improved-code-for-concentric-map.html v2*/
285  Vec2f ToUnitDisk(Vec2f O) {
286  if (O == Vec2f(0, 0))
287  return O;
288  float phi, r;
289  float a = 2 * O(0) - 1;
290  float b = 2 * O(1) - 1;
291  if (a*a > b*b) { // use squares instead of absolute values
292  r = a;
293  phi = (M_PI / 4)*(b / a);
294  }
295  else {
296  r = b;
297  phi = (M_PI / 4)*(a / b) + (M_PI / 2);
298  }
299  return Vec2f(r*cos(phi), r*sin(phi));
300  }

302  /* http://psgraphics.blogspot.de/2011/01/improved-code-for-concentric-map.html v1 */
303  Vec2f to_unit_disk(Vec2f in)
304  {
305  float seedx = in(0);
306  float seedy = in(1);
307 
308  double phi, r = 1.0;
309 
310  double a = 2 * seedx - 1; /* (a,b) is now on [-1,1]^2 */
311  double b = 2 * seedy - 1;
312 
313  if (seedx == 0 && seedy == 0)
314  return Vec2f(0, 0);
315 
316  if (a > -b) { /* region 1 or 2 */
317  if (a > b) { /* region 1, also |a| > |b| */
318  r = a;
319  phi = (M_PI / 4) * (b / a);
320  }
321  else { /* region 2, also |b| > |a| */
322  phi = (M_PI / 4) * (2 - (a / b));
323  }
324  }
325  else { /* region 3 or 4 */
326  if (a < b) { /* region 3, also |a| >= |b|, a != 0 */
327  r = -a;
328  phi = (M_PI / 4) * (4 + (b / a));
329  }
330  else { /* region 4, |b| >= |a|, but a==0 and b==0 could occur. */
331  r = -b;
332  if (b != 0)
333  phi = (M_PI / 4) * (6 - (a / b));
334  else
335  phi = 0;
336  }
337  }
338  return Vec2f(r * cos(phi), r * sin(phi));
339  }

349  Vec2f ConcentricSampleDisk(const Vec2f &u) {
350  // Map uniform random numbers to $[-1,1]^2$
351  Vec2f uOffset = 2.f * u - Vec2f(1, 1);
352 
353  // Handle degeneracy at the origin
354  if (uOffset(0) == 0 && uOffset(1) == 0) return Vec2f(0, 0);
355 
356  // Apply concentric mapping to point
357  float theta, r;
358  if (std::abs(uOffset(0)) > std::abs(uOffset(1))) {
359  r = uOffset(0);
360  theta = (M_PI / 4) * (uOffset(1) / uOffset(0));
361  }
362  else {
363  r = uOffset(1);
364  theta = (M_PI / 4) - (M_PI / 4) * (uOffset(0) / uOffset(1));
365  }
366  return r * Vec2f(std::cos(theta), std::sin(theta));
367  }

455  Color getRainbowColor(float max, float input)

462  float b = 0.0f;

470  else if (2 <= x && x <= 3) b = x - 2;
471  else if (5 <= x && x <= 6) b = 1.0f - (x - 5);
